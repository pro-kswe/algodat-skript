\section{Bubblesort}

\begin{lstlisting}[language={pseudocode}, caption={Bubblesort}, label={lst-algo-bubblesort}]
input: Liste mit n Zahlen

i = 0
while i < n - 1:
  j = 0
  while j < n - i - 1:
    if L[j] > L[j+1] then:
      % Tausche Elemente
      tmp = L[j]
      L[j] = L[j+1]
      L[j+1] = tmp
    j = j + 1
  i = i + 1
  
output: Sortierte Liste mit n Zahlen
\end{lstlisting}
